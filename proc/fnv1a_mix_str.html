   subroutine fnv1a_mix_str(h, s)
      integer(int64), intent(inout) :: h
      character(len=*), intent(in)  :: s
      integer :: i
      do i = 1, len_trim(s)
         call fnv1a_mix_i64(h, int(iachar(s(i:i)), int64))
      end do
   end subroutine fnv1a_mix_str
