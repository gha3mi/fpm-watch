   subroutine log_info(w, msg)
      type(watch_opts_t), intent(in) :: w
      character(len=*), intent(in)   :: msg
      if (w%verbosity < 0) return
      write(output_unit,'(a)') &
         colorize("info       |", color_fg='blue_intense', style='bold_on') // "  " // trim(msg)
   end subroutine log_info
